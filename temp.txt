package com.example.thymeleaf;

import org.junit.Test;
import org.junit.runner.RunWith;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.junit4.SpringRunner;

import java.util.ArrayList;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

@RunWith(SpringRunner.class)
@SpringBootTest
public class ThymeleafApplicationTests {
	private Set<List> set = new HashSet<>();


	@Test
	public void contextLoads() {
		List<Integer> list = new ArrayList();
		list.add(1);
		list.add(2);
		list.add(3);
		list.add(4);
		list.add(5);
		list.add(6);
		list.add(7);
		int constSum = 7;
		getRadomSet(list,constSum);
		System.out.print(set.size());
	}
	private void getRadomSet(List<Integer> list,int constSum){
		List<Integer> result = new ArrayList<>();
		for(int i=0; i<=list.size()-1;  i++ ){
			result.add(list.get(i));
			int tempSum = constSum - list.get(i);
			List<Integer> rightList = new ArrayList<>();
			if(i+1 <= list.size()){
				rightList.addAll(i+1,list);
			}
			if(rightList.contains(tempSum)){
				result.add(tempSum);
				set.add(result);
				if(i <= list.size()-2){
					result.remove(result.size()-1);
					rightList.remove(rightList.indexOf(tempSum));
					getRadomSet(rightList,tempSum);
				}
			}else{
				if(rightList.size() != 0){
					getRadomSet(rightList,tempSum);
				}
				return;
			}
		}

	}

}
